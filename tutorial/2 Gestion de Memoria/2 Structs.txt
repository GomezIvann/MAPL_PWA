' Estructuras o Tipos de usuario:
' 		(#type || #struct) IDENT_S ':' '{' IDENT_P: tipo '}'
' OJO, a diferencia de los tipos primitivos, las estructuras son sensibles a mayúsculas y minúsculas. 
' Ej:
' 	#type Persona: {...} la variable:
'		- alberto:Persona CORRECTO
'		- alberto:persona INCORRECTO

#type Persona: {
	sueldo:real
	edad:int	
	inicial:char
}

#global	a:byte
#global alberto:Persona
#global javi:Persona

call main
halt

main:
	' alberto.edad = 20;
	pusha 1
	push 4
	add
	push 20
	store 

	' javi.edad = 20;
	pusha 8
	push 4
	add
	push 20
	store 

	' alberto.inicial = 'a';
	pusha 1
	push 6
	add
	pushb 97
	storeb

	' javi.inicial = 'a';
	pusha 8
	push 6
	add
	pushb 97
	storeb


	' alberto.sueldo = 3000.0;
	pusha 1
	push 0
	add
	pushf 3000.0
	storef

	' javi.sueldo = 3000.0;
	pusha 8
	push 0
	add
	pushf 3000.0
	storef

ret
